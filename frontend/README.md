## Clone the repository
git clone https://github.com/ThulaniDissanayake/seat_reservation.git
cd seat_reservation


## Setup Backend
cd backend
npm install


## Create .env with
PORT=5000
JWT_SECRET=SecurityKeyJWT
DB_HOST=localhost
DB_USER=seatuser
DB_PASSWORD=
DB_NAME=seat_reservation_db


## Start backend
npm run dev


## Setup Frontend
cd frontend
npm install
npm start
